The definition of minus1Trunc, the (-1)-truncation. Here is what it would look like if Coq supported higher inductive types natively: 
   Inductive minus1Trunc (A : Type) : Type :=
   | min1 : A -> minus1Trunc A
   | min1_path : forall (x y: minus1Trunc A), x = y

Private Inductive minus1Trunc (A :Type) : Type :=
min1 : A minus1Trunc A.
Bind Scope prop_trunc_scope with minus1Trunc.
Axiom min1_path : {A : Type} (x y : minus1Trunc A), x = y.
Arguments min1 {A} a.

Definition minus1Trunc_rect :
{A : Type} {P : minus1Trunc A Type}
(dmin1 : a : A, P (min1 a))
(dpath : {x y : minus1Trunc A} (z : P x) (w : P y), transport _ (min1_path x y) z = w)
(x : minus1Trunc A), P x.
intros ???? [x]. apply dmin1.
Defined.

Definition minus1Trunc_compute_min1 : {A : Type} {P : minus1Trunc A Type}
(dmin1 : (a : A), P (min1 a))
(dpath : (x y : minus1Trunc A) (z : P x) (w : P y), transport _ (min1_path x y) z = w),
(a : A), minus1Trunc_rect dmin1 dpath (min1 a) = dmin1 a.
reflexivity.
Defined.

Axiom minus1Trunc_compute_path : {A : Type} {P : minus1Trunc A Type}
(dmin1 : (a : A), P (min1 a))
(dpath : (x y : minus1Trunc A) (z : P x) (w : P y), transport _ (min1_path x y) z = w),
(x y : minus1Trunc A),
apD (minus1Trunc_rect dmin1 dpath) (min1_path x y)
= dpath x y (minus1Trunc_rect dmin1 dpath x) (minus1Trunc_rect dmin1 dpath y).
End minus1Trunc.

get the type of the goal
let G := match goal with |- ?G constr:(G) end in
prove that the goal is an hProp
let H := fresh in
